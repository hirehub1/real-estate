  <script>
    // Util: CSV encode
    function toCsv(obj){
      const keys = Object.keys(obj);
      const head = keys.join(",");
      const row = keys.map(k => '"' + String(obj[k] ?? '').replaceAll('"','""') + '"').join(',');
      return head + "\n" + row;
    }

    // Year
    document.getElementById('year').textContent = new Date().getFullYear();

    // Submit handler (local capture + download buttons)
    function handleSubmit(e){
      e.preventDefault();
      const form = e.target;
      if(!form.reportValidity()) return;

      const data = Object.fromEntries(new FormData(form).entries());
      data.country = 'Russia';
      data.kind = 'work';
      data.timestamp = new Date().toISOString();

      // Show success + enable downloads
      const jsonStr = JSON.stringify(data, null, 2);
      const csvStr = toCsv(data);

      document.getElementById('workSuccess').classList.remove('hidden');
      enableDownload('workDownloadJson', jsonStr, 'work-application.json', 'application/json');
      enableDownload('workDownloadCsv', csvStr, 'work-application.csv', 'text/csv');
    }

    function enableDownload(btnId, content, filename, mime){
      const btn = document.getElementById(btnId);
      const blob = new Blob([content], {type: mime});
      const url = URL.createObjectURL(blob);
      btn.disabled = false;
      btn.onclick = () => {
        const a = document.createElement('a');
        a.href = url; a.download = filename; a.click();
        setTimeout(()=>URL.revokeObjectURL(url), 2000);
      };
    }

    document.getElementById('workForm').addEventListener('submit', handleSubmit);
  </script>
